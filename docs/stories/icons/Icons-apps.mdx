# Apps and Connectors Icons

This set of icons uses a SVG Sprite and should already be included with `ode-bootstrap`.

## Usage

Use this set of icons with the `AppCard` Component.

```jsx
import { AppIcon } from "@ode-react-ui/components";

export default function App() {
  // AppCard has its own context to pass "app" information to AppIcon

  return (
    <AppCard
      app={{
        address: "/blog",
        icon: "blog-large",
        name: "Blog",
        scope: [],
        display: false,
        displayName: "",
        isExternal: false,
      }}
      isHeading
      headingStyle="h3"
      as="h1"
    >
      <AppIcon size="48" />
    </AppCard>
  );
}
```
